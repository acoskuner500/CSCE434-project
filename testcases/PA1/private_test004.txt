// Int Arithmetic
main

int a, b, c;

{
    let a = 5;
    call printInt(a);

    let a++;
    call printInt(a);

    let b = (a - 1) % 4;
    call printInt(b);

    let c = a + b;
    call printInt(c);

    let c %= 5;
    call printInt(c);

    let c--;
    call printInt(c);

    let c = 2;
    let c *= (a + b) / (3 * c);
    call printInt(c);

    let c *= (a + b) ^ 6;
    call printInt(c);
}.
